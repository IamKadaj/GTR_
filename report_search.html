function searchReport(event) { event.preventDefault(); const searchInput = document.getElementById("search-input"); const query = searchInput.value.trim(); if (!query) { alert("Please enter a search query."); return; } const encodedQuery = encodeURIComponent(query); const url = `https://transparencyreport.google.com/copyright/explore?copyright_data_exploration=q:${encodedQuery};ce:owner;size:10&lu=copyright_data_exploration`; // Navigate to the search result page instead of opening it in a new tab window.location.href = url; }
